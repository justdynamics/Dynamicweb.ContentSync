    **3. Update SaveSyncSettingsCommand.Handle()** to persist new fields and add OutputDirectory validation (per D-05, D-07):

    Per D-05, OutputDirectory must exist on disk relative to the host's `wwwroot/Files/System` directory. The config file lives under `wwwroot/Files/` (see ConfigPathResolver.CandidatePaths). To resolve the Files/System base path:
    - Get the config file path from `ConfigPathResolver.FindOrCreateConfigFile()`
    - The config file is at `wwwroot/Files/ContentSync.config.json`
    - Navigate from the config file's directory up to `Files/`, then into `System/` to get the Files/System base path
    - Combine that with OutputDirectory and check `Directory.Exists`

    ```csharp
    public override CommandResult Handle()
    {
        if (Model is null)
            return new() { Status = CommandResult.ResultType.Invalid, Message = "Model data must be given" };

        if (string.IsNullOrWhiteSpace(Model.OutputDirectory))
            return new() { Status = CommandResult.ResultType.Invalid, Message = "Output Directory is required" };

        try
        {
            var configPath = ConfigPathResolver.FindOrCreateConfigFile();

            // Per D-05: Validate OutputDirectory exists on disk relative to Files/System
            var filesDir = Path.GetDirectoryName(configPath)!; // wwwroot/Files/
            var systemDir = Path.Combine(filesDir, "System");
            var resolvedOutputDir = Path.GetFullPath(
                Path.Combine(systemDir, Model.OutputDirectory.TrimStart('\\', '/')));

            if (!Directory.Exists(resolvedOutputDir))
                return new()
                {
                    Status = CommandResult.ResultType.Invalid,
                    Message = $"Output Directory does not exist: {Model.OutputDirectory} (resolved to {resolvedOutputDir})"
                };

            var existingConfig = ConfigLoader.Load(configPath);

            var conflictStrategy = Model.ConflictStrategy switch
            {
                "source-wins" => Configuration.ConflictStrategy.SourceWins,
                _ => Configuration.ConflictStrategy.SourceWins
            };

            var updatedConfig = new SyncConfiguration
            {
                OutputDirectory = Model.OutputDirectory,
                LogLevel = Model.LogLevel,
                DryRun = Model.DryRun,
                ConflictStrategy = conflictStrategy,
                Predicates = existingConfig.Predicates
            };

            ConfigWriter.Save(updatedConfig, configPath);

            return new() { Status = CommandResult.ResultType.Ok, Model = Model };
        }
        catch (InvalidOperationException ex)
        {
            return new() { Status = CommandResult.ResultType.Error, Message = ex.Message };
        }
    }
    ```

    Key points for D-05 implementation:
    - `configPath` points to `wwwroot/Files/ContentSync.config.json`
    - `Path.GetDirectoryName(configPath)` gives the `wwwroot/Files/` directory
    - `Path.Combine(filesDir, "System")` gives the `wwwroot/Files/System/` base
    - `Model.OutputDirectory` is joined to that base (e.g., `\System\ContentSync` becomes `wwwroot/Files/System/System/ContentSync`)
    - `TrimStart('\\', '/')` strips leading separators so `Path.Combine` works correctly
    - If the resolved path does not exist on disk, return `Invalid` with a descriptive message showing both the input and resolved path
