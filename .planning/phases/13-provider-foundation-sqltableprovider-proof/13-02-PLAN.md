```csharp
// SanitizeFolderName pattern:
private static string SanitizeFolderName(string name)
{
    var trimmed = name.Trim();
    if (string.IsNullOrEmpty(trimmed)) return "_unnamed";
    var invalid = Path.GetInvalidFileNameChars();
    return string.Concat(trimmed.Select(c => invalid.Contains(c) ? '_' : c));
}
```

Since `XmlDataGroupRepository` is `internal` (Pitfall 1), parse DataGroup XML directly using `System.Xml.Linq`. The XML files live at `Files/System/Deployment/DataGroups/` and can be found by enumerating .xml files.

```csharp
namespace Dynamicweb.ContentSync.Providers.SqlTable;

using System.Xml.Linq;
using Dynamicweb.ContentSync.Models;

public class DataGroupMetadataReader
{
    private readonly ISqlExecutor _sqlExecutor;
    private readonly string _dataGroupsPath;

    public DataGroupMetadataReader(ISqlExecutor sqlExecutor, string dataGroupsPath)
    {
        _sqlExecutor = sqlExecutor;
        _dataGroupsPath = dataGroupsPath;
    }
```

Method `GetTableMetadata(string dataGroupId)`:
1. Find XML file: enumerate all `.xml` files in `_dataGroupsPath` recursively, parse each looking for `<DataGroup Id="{dataGroupId}">`. Cache parsed results.
2. Extract first `DataItemType` element whose `ProviderTypeName` contains `SqlDataItemProvider`
3. Read `ProviderParameters`: `Table`, `NameColumn`, `CompareColumns` from `<Parameter Name="..." Value="..." />` elements
4. Query PK columns: Use `ISqlExecutor.ExecuteReader` with CommandBuilder for `sp_pkeys @table_name = '{tableName}'` — read `COLUMN_NAME` from results, sort alphabetically (per D-11)
5. Query identity columns: Use `ISqlExecutor.ExecuteReader` with CommandBuilder for `SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = @table AND COLUMNPROPERTY(OBJECT_ID(TABLE_SCHEMA + '.' + TABLE_NAME), COLUMN_NAME, 'IsIdentity') = 1`
6. Query all columns: Use `ISqlExecutor.ExecuteReader` with CommandBuilder for `SELECT TOP 0 * FROM [{tableName}]` then read column names from `reader.GetSchemaTable()` or iterate `reader.FieldCount`/`reader.GetName(i)`
7. Return `TableMetadata` record

If DataGroup XML not found, throw `InvalidOperationException($"DataGroup '{dataGroupId}' not found in {_dataGroupsPath}")`.

- Build string: `"{KEY}={VALUE}|{KEY}={VALUE}"` with key uppercased
- Return MD5 hash via `System.Security.Cryptography.MD5.HashData` and convert to hex string (or use a helper). Do NOT reference `StringHelper.Md5HashToString` as it may be internal. Implement: `Convert.ToHexString(MD5.HashData(Encoding.UTF8.GetBytes(s))).ToLowerInvariant()`

```csharp
namespace Dynamicweb.ContentSync.Providers.SqlTable;

public class SqlTableProvider : SerializationProviderBase
{
    private readonly DataGroupMetadataReader _metadataReader;
    private readonly SqlTableReader _tableReader;
    private readonly FlatFileStore _fileStore;

    public override string ProviderType => "SqlTable";
    public override string DisplayName => "SQL Table Provider";

    public SqlTableProvider(DataGroupMetadataReader metadataReader, SqlTableReader tableReader, FlatFileStore fileStore)
    {
        _metadataReader = metadataReader;
        _tableReader = tableReader;
        _fileStore = fileStore;
    }
```

`Serialize` method:
1. Validate predicate has `DataGroupId` (call `ValidatePredicate` first, throw if invalid)
2. `var metadata = _metadataReader.GetTableMetadata(predicate.DataGroupId!)`
3. Log: `$"Serializing table {metadata.TableName} (DataGroup: {predicate.DataGroupId})"`
4. `var rows = _tableReader.ReadAllRows(metadata.TableName).ToList()`
5. Log: `$"Read {rows.Count} rows from {metadata.TableName}"`
6. `_fileStore.WriteMeta(outputRoot, metadata.TableName, metadata)`
7. For each row: generate identity via `_tableReader.GenerateRowIdentity(row, metadata)`, then `_fileStore.WriteRow(outputRoot, metadata.TableName, identity, row)`
8. Return `SerializeResult` with `RowsSerialized = rows.Count, TableName = metadata.TableName`

`ValidatePredicate` method:
- If `predicate.ProviderType != "SqlTable"`, return invalid with error "Provider type mismatch"
- If `predicate.DataGroupId` is null or empty, return invalid with error "DataGroupId is required for SqlTable predicates" (per D-05)
- Otherwise return valid
